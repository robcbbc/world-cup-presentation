/** 
 * @test
 */
public function itShouldPresentKnockoutFixtures()
{
    $matches = $this->stubMatches(
        array(
            array('getGroupName' => 'Group A', 'isGroup' => true),
            ...
        )
    );

    $tables = $this->stubTables(array());

    $this->presenter->present();

    // Verifies there are three groups
    $this->assertEquals(2, count($this->presenter->knockout));

    // Verifies the number of fixtures per group
    $this->assertEquals(2, count($this->presenter->knockout[0]['fixtures']));
    $this->assertEquals(1, count($this->presenter->knockout[1]['fixtures']));

    //Should order groups by table, not by fixture
    $this->assertEquals('Last 16', $this->presenter->knockout[0]['name']);
    $this->assertEquals('Semi Final', $this->presenter->knockout[1]['name']);

    //Verifies render was called on matches
    $this->verifyRenders($matches, $tables);
}
